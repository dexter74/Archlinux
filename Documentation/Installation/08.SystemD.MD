```bash
NETWORK_WIFI_INTERFACE=wlan0
```

### II. Désactivation des services
```bash
clear ;

################################################################################################################
systemctl disable NetworkManager ;

systemctl disable wpa_supplicant@wlan0 ;

systemctl disable iwd ;
systemctl disable ntpd ;
```
