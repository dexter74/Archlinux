**Réseau**

ETH0
```
echo "[Match]
Name=eth0

[Link]
MTUBytes=1500

[Network]
DHCP=ipv4
Address=192.168.1.108/24
Gateway=192.168.1.1
DNS=192.168.1.1
DNS=192.168.1.40" > /etc/systemd/network/20-dhcp.network ;
systemctl restart systemd-networkd.service ;
```


WLAN0
```
echo "[Match]
Name=wlan0

[Network]
DHCP=Yes" > /etc/systemd/network/10-static-eth0.network ;
```
