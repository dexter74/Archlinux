```
echo "
[Unit]
Description=Additional drive

[Mount]
What=/dev/sdc7
Where=/mnt/data
Type=ntfs
Options=defaults

[Install]
WantedBy=multi-user.target" > /etc/systemd/system/mnt-data.mount ;
```

```
systemctl daemon-reload ;
systemctl restart mnt-data.mount ;
```
